#include <stdio.h>

/* (a) Gross salary */
void gross_salary() {
    float basic, da, hra, gross;
    printf("\n(a) Gross Salary\n");
    printf("Enter basic salary: ");
    scanf("%f", &basic);

    da = 0.40f * basic;
    hra = 0.20f * basic;
    gross = basic + da + hra;

    printf("Gross salary = %.2f\n", gross);

    /* Sample Output:
       Enter basic salary: 10000
       Gross salary = 16000.00
    */
}

/* (b) Distance conversion */
void distance_conversion() {
    float km;
    printf("\n(b) Distance Conversion\n");
    printf("Enter distance in kilometers: ");
    scanf("%f", &km);

    printf("Meters      = %.2f\n", km * 1000);
    printf("Feet        = %.2f\n", km * 3280.84);
    printf("Inches      = %.2f\n", km * 39370.08);
    printf("Centimeters = %.2f\n", km * 100000);

    /* Sample Output:
       Enter distance in kilometers: 2
       Meters      = 2000.00
       Feet        = 6561.68
       Inches      = 78740.16
       Centimeters = 200000.00
    */
}

/* (c) Aggregate and percentage */
void aggregate_percentage() {
    int m1, m2, m3, m4, m5, agg;
    float percent;

    printf("\n(c) Aggregate and Percentage\n");
    printf("Enter marks in 5 subjects: ");
    scanf("%d%d%d%d%d", &m1, &m2, &m3, &m4, &m5);

    agg = m1 + m2 + m3 + m4 + m5;
    percent = agg / 5.0f;

    printf("Aggregate = %d\n", agg);
    printf("Percentage = %.2f%%\n", percent);

    /* Sample Output:
       Enter marks in 5 subjects: 78 69 88 91 74
       Aggregate = 400
       Percentage = 80.00%
    */
}

/* (d) Fahrenheit to Centigrade */
void fahrenheit_to_centigrade() {
    float f, c;
    printf("\n(d) Fahrenheit to Centigrade\n");
    printf("Enter temperature in Fahrenheit: ");
    scanf("%f", &f);

    c = (f - 32) * 5 / 9;

    printf("Temperature in Centigrade = %.2f\n", c);

    /* Sample Output:
       Enter temperature in Fahrenheit: 98.6
       Temperature in Centigrade = 37.00
    */
}

/* (e) Rectangle and circle */
void rectangle_circle() {
    float l, b, r, pi = 3.14159f;
    float area_rect, peri_rect, area_circle, circum_circle;

    printf("\n(e) Rectangle and Circle\n");
    printf("Enter length, breadth and radius: ");
    scanf("%f%f%f", &l, &b, &r);

    area_rect = l * b;
    peri_rect = 2 * (l + b);
    area_circle = pi * r * r;
    circum_circle = 2 * pi * r;

    printf("Rectangle area = %.2f\n", area_rect);
    printf("Rectangle perimeter = %.2f\n", peri_rect);
    printf("Circle area = %.2f\n", area_circle);
    printf("Circle circumference = %.2f\n", circum_circle);

    /* Sample Output:
       Enter length, breadth and radius: 5 4 3
       Rectangle area = 20.00
       Rectangle perimeter = 18.00
       Circle area = 28.27
       Circle circumference = 18.85
    */
}

/* (f) Interchange two numbers */
void interchange_numbers() {
    int c, d, temp;
    printf("\n(f) Interchange Two Numbers\n");
    printf("Enter C and D: ");
    scanf("%d%d", &c, &d);

    temp = c;
    c = d;
    d = temp;

    printf("After interchange:\n");
    printf("C = %d\n", c);
    printf("D = %d\n", d);

    /* Sample Output:
       Enter C and D: 10 20
       After interchange:
       C = 20
       D = 10
    */
}

/* (g) Sum of digits of a five-digit number */
void sum_of_digits() {
    int n, sum = 0, digit;
    printf("\n(g) Sum of Digits\n");
    printf("Enter a five-digit number: ");
    scanf("%d", &n);

    while (n > 0) {
        digit = n % 10;
        sum += digit;
        n /= 10;
    }

    printf("Sum of digits = %d\n", sum);

    /* Sample Output:
       Enter a five-digit number: 12345
       Sum of digits = 15
    */
}

/* (h) Reverse a five-digit number */
void reverse_number() {
    int n, rev = 0, digit;
    printf("\n(h) Reverse Number\n");
    printf("Enter a five-digit number: ");
    scanf("%d", &n);

    while (n > 0) {
        digit = n % 10;
        rev = rev * 10 + digit;
        n /= 10;
    }

    printf("Reversed number = %d\n", rev);

    /* Sample Output:
       Enter a five-digit number: 12345
       Reversed number = 54321
    */
}

/* (i) Sum of first and last digit of a four-digit number */
void sum_first_last_digit() {
    int n, first, last;
    printf("\n(i) Sum of First and Last Digit\n");
    printf("Enter a four-digit number: ");
    scanf("%d", &n);

    last = n % 10;
    first = n;
    while (first >= 10) {
        first /= 10;
    }

    printf("Sum of first and last digit = %d\n", first + last);

    /* Sample Output:
       Enter a four-digit number: 1234
       Sum of first and last digit = 5
    */
}

/* (j) Illiterate men and women */
void illiterate_men_women() {
    int population = 80000;
    int men, women, literate, literate_men;
    int illiterate_men, illiterate_women, literate_women;

    printf("\n(j) Illiterate Men and Women\n");

    men = 52 * population / 100;
    women = population - men;
    literate = 48 * population / 100;
    literate_men = 35 * population / 100;

    illiterate_men = men - literate_men;
    literate_women = literate - literate_men;
    illiterate_women = women - literate_women;

    printf("Illiterate men = %d\n", illiterate_men);
    printf("Illiterate women = %d\n", illiterate_women);

    /* Sample Output:
       Illiterate men = 13600
       Illiterate women = 28000
    */
}

    printf("\n(l) Cost Price of One Item\n");

    /* Sample Output:

       Enter total profit: 300
       Cost price of one item = 100.00
    */

/* (m) Add one to each digit */
void add_one_each_digit() {
    int n, digit, result = 0, place = 1;

    printf("\n(m) Add One to Each Digit\n");
    printf("Enter a five-digit number: ");
    scanf("%d", &n);

    while (n > 0) {
        digit = n % 10;
        digit = (digit + 1) % 10;
        result += digit * place;
        place *= 10;
        n /= 10;
    }

    printf("New number = %d\n", result);

    /* Sample Output:
       Enter a five-digit number: 12391
       New number = 23402
    */
}
